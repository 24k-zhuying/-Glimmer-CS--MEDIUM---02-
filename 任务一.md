#include <stdio.h>
#include <math.h>
#include <string.h>

//将二进制转化为十进制的函数
//two是传入的二进制字符串,resultNums是最后计算结果
void twoToTen(char *two, char *resultNums) {
    int len = strlen(two);
    int dotPos = -1;
    // 查找小数点位置
    for (int i = 0; i < len; i++) {
        if (two[i] == '.'){
            dotPos = i;
            break;
        }
    }
    double ten = 0;
    if (dotPos == -1) {
        // 如果没有小数点，按照整数转换
        for (int i = 0; len - i - 1 >= 0; i++) {
            if (two[i] == '1') {
                ten += pow(2, len - i - 1);
            }
        }
    } else {
        // 处理整数部分
        for (int i = 0; i < dotPos; i++) {
            if (two[i] == '1') {
                ten += pow(2, dotPos - i - 1);
            }
        }
        // 处理小数部分
        for (int i = dotPos + 1; i < len; i++) {
            if (two[i] == '1') {
                ten += pow(2, dotPos - i);
            }
        }
    }
    //将结果保存在字符串中
    sprintf(resultNums,"%f",ten);

    return;
}

//十进制转二进制函数
//ten是传入的十进制数，result是最后计算结果
void tenToTwo(double ten, char *result) {
    int zhengShuPart = (int)ten;
    double xiaoShuPart = ten - zhengShuPart;
    int index = 0;
    int temp;

    //处理整数部分
    if (zhengShuPart == 0) {
        result[index++] = '0';
    } else {
        //将整数部分不断除以2取余，得到二进制表示
        while (zhengShuPart > 0) {
            temp = zhengShuPart % 2;
            result[index++] = temp + '0';
            zhengShuPart = zhengShuPart / 2;
        }
        //反转整数部分的二进制字符串
        int len = index;
        for (int i = 0; i < len / 2; i++) {
            char temp = result[i];
            result[i] = result[len - i - 1];
            result[len - i - 1] = temp;
        }
    }
    //如果有小数部分则处理小数部分
    if (xiaoShuPart > 0) {
        result[index++] = '.';
        int count = 0;
        //不断将小数部分乘以2取整，得到二进制表示
        while (xiaoShuPart > 0 && count < 32) {
            xiaoShuPart = xiaoShuPart * 2;
            if (xiaoShuPart >= 1) {
                result[index++] = '1';
                xiaoShuPart = xiaoShuPart - 1;
            } else {
                result[index++] = '0';
            }
            count++;
        }
    }
    result[index] = '\0';
}
